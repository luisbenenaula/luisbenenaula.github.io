        function calculateResults() {

            const form = document.getElementById('personalityTest');

            const results = {

                openness: (parseInt(form.q1.value) + parseInt(form.q2.value) + parseInt(form.q3.value)) / 3,

                conscientiousness: (parseInt(form.q4.value) + parseInt(form.q5.value) + parseInt(form.q6.value)) / 3,

                extraversion: (parseInt(form.q7.value) + parseInt(form.q8.value) + parseInt(form.q9.value)) / 3,

                agreeableness: (parseInt(form.q10.value) + parseInt(form.q11.value) + parseInt(form.q12.value)) / 3,

                neuroticism: (parseInt(form.q13.value) + parseInt(form.q14.value) + parseInt(form.q15.value)) / 3

            };


            const labels = ["Apertura a la experiencia", "Conciencia", "Extraversión", "Amabilidad", "Neuroticismo"];

            const data = [results.openness, results.conscientiousness, results.extraversion, results.agreeableness, results.neuroticism];


            const ctx = document.getElementById('resultChart').getContext('2d');

            new Chart(ctx, {

                type: 'radar',

                data: {

                    labels: labels,

                    datasets: [{

                        label: 'Tus Resultados',

                        data: data,

                        backgroundColor: 'rgba(106, 17, 203, 0.2)',

                        borderColor: '#6a11cb',

                        borderWidth: 2

                    }]

                },

                options: {

                    scales: {

                        r: {

                            angleLines: {

                                display: true,

                                color: '#333'

                            },

                            suggestedMin: 0,

                            suggestedMax: 5,

                            ticks: {

                                display: false

                            }

                        }

                    }

                }

            });


            let interpretationHTML = '';


            if (results.openness >= 4) {

                interpretationHTML += '<li><strong>Apertura a la experiencia:</strong> Tienes un alto nivel de Apertura a la experiencia, lo que sugiere que eres una persona muy creativa, curiosa y dispuesta a probar cosas nuevas.</li>';

            } else if (results.openness <= 2) {

                interpretationHTML += '<li><strong>Apertura a la experiencia:</strong> Tu nivel de Apertura a la experiencia es bajo, lo que indica que prefieres la rutina y te sientes más cómodo/a con lo conocido.</li>';

            } else {

                interpretationHTML += '<li><strong>Apertura a la experiencia:</strong> Tu nivel de Apertura a la experiencia es moderado, lo que sugiere que disfrutas de nuevas experiencias pero también valoras la estabilidad.</li>';

            }


            if (results.conscientiousness >= 4) {

                interpretationHTML += '<li><strong>Conciencia:</strong> Eres altamente consciente, lo que significa que eres organizado/a, responsable y tiendes a seguir un enfoque metódico para alcanzar tus objetivos.</li>';

            } else if (results.conscientiousness <= 2) {

                interpretationHTML += '<li><strong>Conciencia:</strong> Tu nivel de Conciencia es bajo, lo que sugiere que puedes ser más flexible y menos estructurado/a en tu enfoque de las tareas.</li>';

            } else {

                interpretationHTML += '<li><strong>Conciencia:</strong> Tu nivel de Conciencia es moderado, lo que indica un buen equilibrio entre organización y flexibilidad.</li>';

            }


            if (results.extraversion >= 4) {

                interpretationHTML += '<li><strong>Extraversión:</strong> Tienes un alto nivel de Extraversión, lo que significa que eres sociable, enérgico/a y disfrutas de estar rodeado/a de otras personas.</li>';

            } else if (results.extraversion <= 2) {

                interpretationHTML += '<li><strong>Extraversión:</strong> Tu nivel de Extraversión es bajo, lo que indica que prefieres la tranquilidad y puedes disfrutar más de actividades solitarias.</li>';

            } else {

                interpretationHTML += '<li><strong>Extraversión:</strong> Tu nivel de Extraversión es moderado, lo que sugiere que disfrutas tanto de la compañía de otros como de momentos de soledad.</li>';

            }


            if (results.agreeableness >= 4) {

                interpretationHTML += '<li><strong>Amabilidad:</strong> Tu nivel de Amabilidad es alto, lo que significa que eres cooperativo/a, empático/a y te preocupas por los demás.</li>';

            } else if (results.agreeableness <= 2) {

                interpretationHTML += '<li><strong>Amabilidad:</strong> Tu nivel de Amabilidad es bajo, lo que sugiere que puedes ser más competitivo/a o independiente en tus interacciones.</li>';

            } else {

                interpretationHTML += '<li><strong>Amabilidad:</strong> Tu nivel de Amabilidad es moderado, lo que indica un buen equilibrio entre empatía y asertividad.</li>';

            }


            if (results.neuroticism >= 4) {

                interpretationHTML += '<li><strong>Neuroticismo:</strong> Tienes un alto nivel de Neuroticismo, lo que sugiere que tiendes a experimentar emociones negativas como ansiedad o estrés con mayor frecuencia.</li>';

            } else if (results.neuroticism <= 2) {

                interpretationHTML += '<li><strong>Neuroticismo:</strong> Tu nivel de Neuroticismo es bajo, lo que indica que tiendes a ser emocionalmente estable y te mantienes calmado/a en situaciones de presión.</li>';

            } else {

                interpretationHTML += '<li><strong>Neuroticismo:</strong> Tu nivel de Neuroticismo es moderado, lo que sugiere que puedes experimentar estrés ocasionalmente, pero generalmente te recuperas rápidamente.</li>';

            }


            document.getElementById('interpretation').innerHTML = interpretationHTML;

        }
